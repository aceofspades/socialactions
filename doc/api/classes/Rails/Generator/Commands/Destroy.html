422:           destination = destination_path(relative_destination)
423:           if File.exist?(destination)
424:             logger.rm relative_destination
425:             unless options[:pretend]
426:               if options[:svn]
427:                 # If the file has been marked to be added
428:                 # but has not yet been checked in, revert and delete
429:                 if options[:svn][relative_destination]
430:                   system("svn revert #{destination}")
431:                   FileUtils.rm(destination)
432:                 else
433:                 # If the directory is not in the status list, it

437:               elsif options[:git]
438:                 if options[:git][:new][relative_destination]
439:                   # file has been added, but not committed
440:                   system("git reset HEAD #{relative_destination}")
441:                   FileUtils.rm(destination)
442:                 elsif options[:git][:modified][relative_destination]
443:                   # file is committed and modified
444:                   system("git rm -f #{relative_destination}")
